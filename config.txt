CREATE TABLE empleados{
    id SERIAL primary key,
    nombre varchar(100) NOT NULL,
    cedula varchar(100) NOT NULL,
    edad INT NOT NULL,
    sexo varchar(10),
    telefono varchar(20),
    cargo(50)
}
